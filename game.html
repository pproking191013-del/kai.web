const box = document.getElementById("box"); const result = document.getElementById("result"); let start = 0; let ready = false; function startGame(){ ready = false; box.classList.remove("active"); box.style.background = "#ff3b3b"; // RED = wait result.textContent = "wait..."; setTimeout(()=>{ box.classList.add("active"); box.style.background = "#22c55e"; // GREEN = click result.textContent = "CLICK!"; start = Date.now(); ready = true; }, Math.random()*2000 + 1000); } box.onclick = ()=>{ if(!ready){ result.textContent = "too early 💀"; return; } const time = Date.now() - start; result.textContent = time + " ms"; startGame(); }; startGame();
