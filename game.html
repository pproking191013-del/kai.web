const box = document.getElementById("box"); const result = document.getElementById("result"); let start = 0; let ready = false; let timeout; /* start new round */ function startGame(){ ready = false; box.style.background = "#ff3b3b"; // RED result.textContent = "wait..."; clearTimeout(timeout); timeout = setTimeout(()=>{ box.style.background = "#22c55e"; // GREEN result.textContent = "CLICK!"; start = Date.now(); ready = true; }, Math.random()*2000 + 1000); } box.onclick = ()=>{ // clicked too early if(!ready){ result.textContent = "too early 💀"; clearTimeout(timeout); setTimeout(startGame, 1000); return; } // correct click const time = Date.now() - start; result.textContent = time + " ms"; ready = false; // short delay before next round setTimeout(startGame, 1000); }; startGame();
